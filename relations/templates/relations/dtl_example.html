For Schleifen Variablen

forloop.counter

    {% for elem in hobbys %}
  • {{ forloop.counter }} {{ elem }}
    • {% empty %}
  • Keine Hobbies
    • {% endfor %}

forloop.counter0

    {% for elem in hobbys %}
  • [{{ forloop.counter0 }}] {{ elem }}
    • {% empty %}
  • Keine Hobbies
    • {% endfor %}

forloop.first und forloop.last

    {% for elem in hobbys %} {% if forloop.first %}
  • Das erste Hobby: {{ elem }}
    • {% elif forloop.last %}
  • Das letzte Hobby: {{ elem }}
    • {% else %}
  • {{ elem }}
    • {% endif %} {% empty %}
  • Keine Hobbies
    • {% endfor %}
